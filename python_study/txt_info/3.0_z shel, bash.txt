# основные команды:
ls -a -l -t - список файлов (включая скрытые, в полной форме, сортировка по дате изменения)
pwd - имя и путь текущей дериктории
touch lol.txt - создать файл
echo "бла бла" > имя.txt
cat text.txt - просмотреть файл
cp text1.txt text2.txt - копирует text1 в text2 с заменой
cp text.txt directory/txt - копируем файл text.txt в directory/txt
mv - перемещение с удалением (+ переименование)
rm - удаление файла 
rm -r - удаление каталога с содержимым
wc - калькуляция слов

# взаимодейсткие между командами:
a > b -  поместить "а" в "b"
a >> b - копировать "а" в "b"
a | b - передаст вывод из "а" в "b"
uniq - удаление смежных дубликатов
grep -i -R -Rl - поиск совпадений (grep Горы горы.txt)? 
	i - без учета регистра
	-R - рекурсивный поиск внутри файлов
	-Rl - рекурсивный поиск только по названиям
sed 's/что_меняем/на_что_меняем/g' объект.txt
	-i - сохраняет изменения в файл
	g в конце - глобал режим (меняет во всей строке), без g меняет 1й вариант в строке 

env - возвращает стандартные параметры (PATH, PS1, )

Шаблоны: ~/.bash_profile
alias name="command -i" - назначение своей команды
export USER="Yasya" - создает переменную USER с значением Yasya (вызов переменной со знаком $ $USER)
export PS1=">> " - изменит значек командной строки с $ на >>
